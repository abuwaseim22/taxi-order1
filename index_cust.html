document.getElementById("bookRideBtn").onclick = () => { let from = document.getElementById("from").value; let to = document.getElementById("to").value; let name = document.getElementById("name").value; let p = Number(document.getElementById("passengers").value); let date = document.getElementById("date").value; let time = document.getElementById("time").value; if(p >= 3) p = 3; if(!from || !to || !name || !p || !date || !time){ return alert("عبّي كل الحقول"); } let total = price(from,to) * p; let orderData = { pickup_location: from, dropoff_location: to, passenger_count: p, pickup_date: date, pickup_time: time, customer_name: name, customer_phone: document.getElementById("phoneLogin").value, price: total, status: "new" }; console.log("Sending order:", orderData); db.ref("orders/order" + Date.now()).set(orderData) .then(() => { console.log("SUCCESS ✅"); alert("تم الحجز 🚕\nالسعر: " + total + " دينار"); }) .catch((error) => { console.error("ERROR ❌", error); alert("فشل الإرسال: " + error.message); }); };
